# Change node version if a .nvmrc is available in current folder 
To add at the end of .bashrc file.

```bash
# automatically change node version if .nvmrc detected
OLD_PWD=""
promptCommand() {
  if [ "$OLD_PWD" != "$PWD" ] ; then
    OLD_PWD="$PWD"
    if [ -e .nvmrc ] ;
      then nvm use;
    fi
  fi
}
export PROMPT_COMMAND=promptCommand
```
